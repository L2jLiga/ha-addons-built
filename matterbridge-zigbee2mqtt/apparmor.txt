#include <tunables/global>

profile matterbridge-zigbee2mqtt flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,

  # Start new profile for service
  /usr/bin/matterbridge cx -> matterbridge,
  /usr/bin/matterbridge.sh cx -> matterbridge,
  /usr/local/lib/node_modules/matterbridge/dist/cli.js cx -> matterbridge,
  /usr/lib/node_modules/npm/bin/npx-cli.js cx -> matterbridge,
  /etc/s6-overlay/s6-rc.d/matterbridge-zigbee2mqtt-installer/run cx -> matterbridge,
  /etc/s6-overlay/s6-rc.d/matterbridge/run cx -> matterbridge,
  /usr/bin/node cx -> matterbridge,
  /usr/bin/npm cx -> matterbridge,
  /usr/bin/npx cx -> matterbridge,
  /usr/sbin/nginx cx -> nginx,

  profile matterbridge flags=(attach_disconnected,mediate_deleted,complain) {
    #include <abstractions/base>
    #include <abstractions/nameservice>

    # Access network
    network,

    # Receive signals from S6-Overlay
    signal (receive) peer=*_matterbridge-zigbee2mqtt,

    # Access to options.json and other files within your addon
    /data/** rw,

    # Access to mapped volumes specified in config.json
    /config/** rw,
    /config/.matterbridge/** rw,

    # Access to libraries
    /lib/libc.musl-x86_64.so.1 r,
    /lib/ld-musl-x86_64.so.1 r,

    # ICU
    /usr/lib/libicudata.so.74 r,
    /usr/lib/libicudata.so.74.1 r,
    /usr/share/icu/74.1/icudt74l.dat r,

    # Access to node_modules
    /usr/{,local/}lib/node_modules rwlixk,
    /usr/{,local/}lib/node_modules/{,**} rwlixk,

    # Access to MatterBridge data
    /root/.matterbridge/** rw,
    /root/MatterBridge/** rw,

    # Access required to start MatterBridge
    /usr/local/lib/node_modules/matterbridge ix,
  }

  profile nginx flags=(attach_disconnected,mediate_deleted,complain) {
    #include <abstractions/base>
    #include <abstractions/openssl>
    #include <abstractions/nameservice>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_matterbridge-zigbee2mqtt,
    
    # Nginx configuration
    /etc/nginx/** r,

    # Access network
    network,

    # Access libraries
    # musl
    /lib/libc.musl-x86_64.so.1 r,
    /lib/ld-musl-x86_64.so.1 r,
    # pcre
    /usr/lib/libpcre** r,
    # libcrypto3
    /{,usr/}lib/libcrypto.so.3 r,
    /usr/lib/engines-3 r,
    /usr/lib/engines-3/** r,
    /usr/lib/ossl-modules r,
    /usr/lib/ossl-modules/** r,
    # libssl3
    /{,usr/}lib//libssl.so.3 r,
    # zlib
    /lib/libz.so** r,

    # Access dev devices
    /dev/null rw,

    # Nginx logs
    /var/lib/nginx rwlk,
    /var/lib/nginx/{,**} rwlk,

    # Nginx PID
    /var/run/nginx.pid rwlk,

    # Executables
    /usr/sbin/nginx rix,
  }
}
